#### 3.2 饼图百分比手动计算（第1061-1111行）

**问题：**
- 原本使用 `params.data._chartData` 存储数据引用
- 这会造成循环引用，导致饼图无法正常渲染

**解决方案：使用闭包变量存储数据**

```javascript
// 初始化饼图
pieChart = echarts.init(pieDom);

// 用于存储当前饼图数据的变量（闭包）
let currentPieData = [];

pieChart.setOption({
    // ... 其他配置
    tooltip: {
        formatter: function(params) {
            // 使用闭包变量计算百分比
            const total = currentPieData.reduce((sum, item) => sum + item.value, 0);
            const percentage = total > 0 ? ((params.value / total * 100).toFixed(2)) : '0.00';
            return `${params.seriesName}<br/>${params.name}: ${params.value.toLocaleString()} (${percentage}%)`;
        }
    },
    series: [{
        label: { 
            formatter: function(params) {
                // 使用闭包变量计算百分比
                const total = currentPieData.reduce((sum, item) => sum + item.value, 0);
                const percentage = total > 0 ? ((params.value / total * 100).toFixed(2)) : '0.00';
                return `${percentage}%`;
            }
        }
    }]
});

// 将 currentPieData 暴露给外部函数使用
pieChart._currentPieData = currentPieData;
```

**为什么不用 `_chartData` 属性？**
- 在数据项上添加 `_chartData: pieData` 会造成循环引用
- ECharts序列化数据时会出错，导致饼图无法渲染
- 使用闭包变量避免了这个问题
